    **E) Text Parsing Function (`parseChallanText`):**

    A pure function defined in the same file (or extracted as a separate utility):

    ```javascript
    function parseChallanText(text, knownPartyNames) {
      const result = { partyName: '', quantity: '', lotNumber: '' };

      // Quantity: find patterns like "123.45 kg" or "100 KGS" or "50Kg"
      const quantityRegex = /(\d+[.,]?\d*)\s*(?:kg|kgs|Kg|KGS|KG)/gi;
      const quantityMatch = quantityRegex.exec(text);
      if (quantityMatch) {
        // Clean the number: replace comma with dot for decimal
        result.quantity = quantityMatch[1].replace(',', '.');
      }

      // Party name: try to match known party names in the text
      const knownNames = knownPartyNames || [];
      for (const name of knownNames) {
        if (text.toLowerCase().includes(name.toLowerCase())) {
          result.partyName = name;
          break;
        }
      }
      // Fallback: look for common challan patterns like "M/s", "Party", "Name"
      if (!result.partyName) {
        const partyRegex = /(?:M\/s|Party|Name|Customer)\s*[:.]?\s*([A-Za-z\s]+)/i;
        const partyMatch = partyRegex.exec(text);
        if (partyMatch) {
          result.partyName = partyMatch[1].trim();
        }
      }

      // Lot number: look for patterns like "Lot No", "Lot #", "Challan No", "Bill No" followed by alphanumeric
      const lotRegex = /(?:Lot\s*(?:No|#|Number|\.)?|Challan\s*(?:No|#|Number|\.)?)\s*[:.]?\s*([A-Za-z0-9\-/]+)/i;
      const lotMatch = lotRegex.exec(text);
      if (lotMatch) {
        result.lotNumber = lotMatch[1].trim();
      }

      return result;
    }
    ```
